Обработка форм (html)

                    
<form name="name" action="index.php" method="post">
  <div>
    <input type="text" name="user[name]" />
    <br />
    <input type="text" name="user[email]" />
    <br />
    <input type="text" name="user[password]" />
    <br />
    <input type="submit" value="Отправить" />
  </div>
</form>
