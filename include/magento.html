<?php

namespace Weblips\Data\Setup;

use Magento\Framework\Setup\InstallSchemaInterface;
use Magento\Framework\Setup\ModuleContextInterface;
use Magento\Framework\Setup\SchemaSetupInterface;
use Magento\Framework\DB\Ddl\Table;
use Magento\Framework\DB\Adapter\AdapterInterface;

class InstallSchema implements InstallSchemaInterface
{
    public function install(SchemaSetupInterface $setup, ModuleContextInterface $context)
    {
        $installer = $setup;
        $installer->startSetup();
        $table = $installer->getConnection()
                ->newTable($installer->getTable('weblips_data'))
                ->addColumn(
                'weblips_id',
                \Magento\Framework\DB\Ddl\Table::TYPE_INTEGER,
                null,
                [
                    'identity' => true,
                    'unsigned' => true,
                    'nullable' => false,
                    'primary' => true
                 ],
                'Weblips Id'
            )
            ->addColumn(
                'name',
                \Magento\Framework\DB\Ddl\Table::TYPE_TEXT,
                250,
                ['nullable' => false],
                'Name'
            )
            ->addColumn(
                'url_key',
                \Magento\Framework\DB\Ddl\Table::TYPE_TEXT,
                250,
                ['nullable' => false],
                'Url Key'
            )
            ->setComment('Weblips Post');
        $installer->endSetup();
    }
}

WHERE:

Здесь мы диспетчерская « checkout_cart_save_before » и « checkout_cart_save_after » события и , таким образом , мы можем легко вызвать какие - либо действия до и после того, как телега сохраняется. Это помогает избежать переписывания всего класса. Magento позволяет перейти значения для наблюдателей , чтобы вызвать действие на возникновение событий , как вы видели в приведенном выше примере.

Для отправки пользовательских событий в Magento 2, создать объект для Magento \ Framework \ Event \ ManagerInterface и передать уникальное имя пользовательского события к объекту.

Отображение событий и наблюдатель:

    
<config xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"xsi:noNamespaceSchemaLocation="urn:magento:framework:Event/etc/events.xsd">
   <event name="sales_order_place_after">
       <observer name="unique_observer_name" instance="Magecomp\MyModule\Observer\MyObserver" />
   </event>
</config>

observer node is having following properties:

  • name – Observer Name that must be unique for each event.
  • instance – Magento class that needs to be executed when the event is dispatched.
  • disabled – To activate or deactivate the observer.
  • shared – экземпляр должен быть создан общий или нет. Default is false.
