Порождающие шаблоны (Creational) — шаблоны проектирования, которые абстрагируют процесс инстанцирования. Они позволяют сделать систему независимой от способа создания, композиции и представления объектов. Шаблон, порождающий классы, использует наследование, чтобы изменять инстанцируемый класс, а шаблон, порождающий объекты, делегирует инстанцирование другому объекту.
Фабричный метод(Fabric Method) не нужно наследовать все и не нужно делать, а потом менять весь функционал предков 
                    
<?php
################## Start for Fabric Method #######################

interface Factory{
    
    public function getProduct();
}

interface Product{
    public function getAll();
}

class BookFactory implements Factory{
    public function getProduct(){
        return new BookProduct();
    }
}

class EBookFactory implements Factory{
    public function getProduct(){
        return new EBookProduct();
    }
}

class BookProduct implements Product {
    public $settings = array('id'=>12, 'name' => 'Hot Summer', 'type'=> 'simple', 'author' => 'Mike Bense' );
    public function getAll(){
        return $this->settings;
    }
}

class EBookProduct implements Product {
    public $settings = array('id'=>12, 'name' => 'EHot Summer', 'type'=> 'simple', 'author' => 'Mike Bense' );
    public function getAll(){
        return $this->settings;
    }
}
###########################################################

$factory = new BookFactory();
$bookProduct = $factory->getProduct();
print_r($bookProduct->getAll());
// OR
$factoryEBook = new EBookFactory();
print_r($factoryEBook->getProduct()->getAll());
